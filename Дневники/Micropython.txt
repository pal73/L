Импорт сторонних библиотек 

Делается с помощью программки в самом модуле

import machine
import network
import time
import mip
station = network.WLAN(network.STA_IF)
station.active(True)
station.connect('systems_promautomatic', 'bCGz37vn') #тут реквизиты сети 
time.sleep(1)
print('Device connected to network: {}'.format(station.isconnected()))
mip.install('github:brainelectronics/micropython-modbus') #тут координаты библиотеки в сети
print('Installation completed')
machine.soft_reset()
